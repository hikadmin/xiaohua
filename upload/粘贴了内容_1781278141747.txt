function generateCalendar() {
  if (!calendarGrid) return;
  
  const firstDay = new Date(currentYear, currentMonth - 1, 1);
  const lastDay = new Date(currentYear, currentMonth, 0);
  const startDayOfWeek = firstDay.getDay();
  const totalDays = lastDay.getDate();
  
  const todayStr = getTodayStr();
  
  let html = '';
  
  for (let i = 0; i < startDayOfWeek; i++) {
    html += '<div class="calendar-day other-month"></div>';
  }
  
  for (let day = 1; day <= totalDays; day++) {
    const dateStr = `${currentYear}-${String(currentMonth).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
    const isToday = dateStr === todayStr;
    const periodInfo = getPeriodInfo(dateStr);
    
    let classes = 'calendar-day';
    
    if (periodInfo.isPeriod) {
      if (periodInfo.isActive) {
        classes += ' period period-active';
      } else {
        classes += ' period';
        if (periodInfo.isStart && periodInfo.isEnd) classes += ' period-single';
        else if (periodInfo.isStart) classes += ' period-start';
        else if (periodInfo.isEnd) classes += ' period-end';
        else classes += ' period-middle';
      }
    }
    
    if (isToday) {
      classes += ' today';
    }
    
    html += `<div class="${classes}" data-date="${dateStr}" onclick="handleDayClick('${dateStr}', ${day}, ${isToday})">${day}</div>`;
  }
  
  calendarGrid.innerHTML = html;
}

function getPeriodInfo(dateStr) {
  let isPeriod = false;
  let isStart = false;
  let isEnd = false;
  let isActive = false;
  
  for (const period of periodData.periods) {
    if (period.start && !period.end) {
      if (dateStr === period.start) {
        isPeriod = true;
        isStart = true;
        isEnd = true;
        isActive = true;
        break;
      }
    } else if (period.start && period.end) {
      if (dateStr >= period.start && dateStr <= period.end) {
        isPeriod = true;
        if (dateStr === period.start) isStart = true;
        if (dateStr === period.end) isEnd = true;
        break;
      }
    }
  }
  
  return { isPeriod, isStart, isEnd, isActive };
}

function hasActivePeriod() {
  for (const period of periodData.periods) {

  return null;
}

}

  }

function extendPeriod(dateStr) {
  const lastPeriod = periodData.periods[periodData.periods.length - 1];
  if (lastPeriod && lastPeriod.end) {
    lastPeriod.end = dateStr;

function cancelActivePeriod() {
  periodData.periods = periodData.periods.filter(p => p.end);
  closeActionSheet();

}

function cancelPeriod(dateStr) {
  periodData.periods = periodData.periods.filter(period => {
    if (period.start && !period.end) {
      if (dateStr === period.start) return false;
    } else if (period.start && period.end) {
      if (dateStr >= period.start && dateStr <= period.end) {
        if (dateStr === period.start && dateStr === period.end) {
          return false;
        } else if (dateStr === period.start) {
          period.start = getNextDay(dateStr);
          if (period.start > period.end) return false;
        } else if (dateStr === period.end) {
          period.end = getPrevDay(dateStr);
          if (period.end < period.start) return false;
        } else {
          periodData.periods.push({ start: period.start, end: getPrevDay(dateStr) });
          periodData.periods.push({ start: getNextDay(dateStr), end: period.end });
          return false;
        }
      }
    }
    return true;
  });
  
  closeActionSheet();
  generateCalendar();
  showToast('已取消该日期的经期记录');
}

  generateCalendar();
}

function saveRecord() {

  document.querySelectorAll('#symptomContainer .symptom-tag.selected').forEach(tag => {
    selectedSymptoms.push(tag.textContent);
  });
  
  const note = document.getElementById('noteInput').value;
  
  const todayStr = getTodayStr();
  
  const newRecord = {
    date: todayStr,
    flow: currentFlow,
    symptoms: selectedSymptoms,
    mood: currentMood,
    note: note
  };
  
  const existingIndex = dailyRecords.findIndex(r => r.date === todayStr);
  if (existingIndex >= 0) {
    dailyRecords[existingIndex] = newRecord;
  } else {
    dailyRecords.unshift(newRecord);

let touchStartX = 0;
let touchEndX = 0;

if (app) {
  app.addEventListener('touchstart', e => {
    touchStartX = e.changedTouches[0].screenX;
  }, false);

  app.addEventListener('touchend', e => {
    touchEndX = e.changedTouches[0].screenX;
    handleSwipe();
  }, false);
}

function handleSwipe() {
  const pages = ['home', 'calendar', 'log', 'profile'];
  const currentIndex = pages.indexOf(currentPage);
  
  if (touchEndX < touchStartX - 50 && currentIndex < pages.length - 1) {
    const nextPage = pages[currentIndex + 1];
    document.querySelector(`[data-page="${nextPage}"]`).click();
  }
  
  if (touchEndX > touchStartX + 50 && currentIndex > 0) {
    const prevPage = pages[currentIndex - 1];
    document.querySelector(`[data-page="${prevPage}"]`).click();
  }
}
