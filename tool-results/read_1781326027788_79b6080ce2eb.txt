   107→   107→function formatDateStr(date: Date): string {
   108→   108→  return `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}-${String(date.getDate()).padStart(2, '0')}`;

   115→   115→function formatShortDate(dateStr: string): string {
   116→   116→  const parts = dateStr.split('-');
   117→   117→  return `${parseInt(parts[1])}月${parseInt(parts[2])}日`;
   118→   118→}
   119→   119→
   120→   120→function parseDate(dateStr: string): Date {
   121→   121→  const parts = dateStr.split('-');
   122→   122→  return new Date(parseInt(parts[0]), parseInt(parts[1]) - 1, parseInt(parts[2]));
   123→   123→}
   124→   124→
   125→   125→function daysBetween(d1: Date, d2: Date): number {
   126→   126→  return Math.round((d2.getTime() - d1.getTime()) / (1000 * 60 * 60 * 24));
   127→   127→}
   128→   128→
   129→   129→function addDays(date: Date, days: number): Date {
   130→   130→  const result = new Date(date);
   131→   131→  result.setDate(result.getDate() + days);
   132→   132→  return result;
   133→   133→}

   257→   257→  function getPeriodInfo(dateStr: string) {
   258→   258→    let isPeriod = false;
   259→   259→    let isStart = false;
   260→   260→    let isEnd = false;
   261→   261→    let isActive = false;
   262→   262→
   263→   263→    for (const period of periods) {
   264→   264→      if (period.startDate && !period.endDate) {
   265→   265→        if (dateStr === period.startDate) {
   266→   266→          isPeriod = true; isStart = true; isEnd = true; isActive = true;
   267→   267→          break;
   268→   268→        }
   269→   269→        // Active period extends from start to today
   270→   270→        const startD = parseDate(period.startDate);
   271→   271→        const checkD = parseDate(dateStr);
   272→   272→        if (checkD > startD && checkD <= new Date()) {
   273→   273→          isPeriod = true;
   274→   274→          if (dateStr === period.startDate) isStart = true;
   275→   275→          isActive = true;
   276→   276→          break;
   277→   277→        }
   278→   278→      } else if (period.startDate && period.endDate) {
   279→   279→        if (dateStr >= period.startDate && dateStr <= period.endDate) {
   280→   280→          isPeriod = true;
   281→   281→          if (dateStr === period.startDate) isStart = true;
   282→   282→          if (dateStr === period.endDate) isEnd = true;
   283→   283→          break;
   284→   284→        }
   285→   285→      }
   286→   286→    }
   287→   287→    return { isPeriod, isStart, isEnd, isActive };
   288→   288→  }

   294→   294→  function getCycleInfo() {
   295→   295→    const cycleLength = profile?.cycleLength || 28;
   296→   296→    const periodLength = profile?.periodLength || 5;
   297→   297→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   298→   298→    const lastPeriod = sortedPeriods[0];
   299→   299→
   300→   300→    if (!lastPeriod) {
   301→   301→      return {
   302→   302→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   303→   303→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: null,
   304→   304→      };
   305→   305→    }
   306→   306→
   307→   307→    const lastStart = parseDate(lastPeriod.startDate);
   308→   308→    const now = new Date();
   309→   309→    now.setHours(0, 0, 0, 0);
   310→   310→    const daysSinceStart = daysBetween(lastStart, now);
   311→   311→
   312→   312→    if (daysSinceStart < 0) {
   313→   313→      return {
   314→   314→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   315→   315→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: lastPeriod.startDate,
   316→   316→      };
   317→   317→    }
   318→   318→
   319→   319→    const dayInCycle = (daysSinceStart % cycleLength) + 1;
   320→   320→    const nextPeriodDate = addDays(lastStart, cycleLength);
   321→   321→    const daysUntilNext = Math.max(0, daysBetween(now, nextPeriodDate));
   322→   322→
   323→   323→    let phase: string;
   324→   324→    if (dayInCycle <= periodLength) {
   325→   325→      phase = 'period';
   326→   326→    } else if (dayInCycle <= 13) {
   327→   327→      phase = 'follicular';
   328→   328→    } else if (dayInCycle <= 16) {
   329→   329→      phase = 'ovulation';
   330→   330→    } else {
   331→   331→      phase = 'luteal';
   332→   332→    }
   333→   333→
   334→   334→    return {
   335→   335→      phase, phaseDay: dayInCycle, daysUntilNext,
   336→   336→      cycleLength, periodLength, nextPeriodDate, lastPeriodStart: lastPeriod.startDate,
   337→   337→    };
   338→   338→  }

   340→   340→  function getPredictedPeriodDays(): string[] {
   341→   341→    const cycleLength = profile?.cycleLength || 28;
   342→   342→    const periodLength = profile?.periodLength || 5;
   343→   343→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   344→   344→    const lastPeriod = sortedPeriods[0];
   345→   345→    if (!lastPeriod || !lastPeriod.endDate) return [];
   346→   346→
   347→   347→    const lastEnd = parseDate(lastPeriod.endDate);

   354→   354→  }

   560→   560→  function generateCalendarDays() {
   561→   561→    const firstDay = new Date(calYear, calMonth - 1, 1);
   562→   562→    const lastDay = new Date(calYear, calMonth, 0);
   563→   563→    const startDayOfWeek = firstDay.getDay();
   564→   564→    const totalDays = lastDay.getDate();
   565→   565→    const todayStr = formatDateStr(new Date());
   566→   566→    const predictedDays = getPredictedPeriodDays();
   567→   567→    const fertileDays = getFertileDays();
   568→   568→
   569→   569→    const days: Array<{
   570→   570→      day: number;
   571→   571→      dateStr: string;
   572→   572→      isToday: boolean;
   573→   573→      isOtherMonth: boolean;
   574→   574→      periodInfo: ReturnType<typeof getPeriodInfo>;
   575→   575→      isPredicted: boolean;
   576→   576→      isFertile: boolean;
   577→   577→    }> = [];
   578→   578→
   579→   579→    for (let i = 0; i < startDayOfWeek; i++) {
   580→   580→      days.push({
   581→   581→        day: 0, dateStr: '', isToday: false, isOtherMonth: true,
   582→   582→        periodInfo: { isPeriod: false, isStart: false, isEnd: false, isActive: false },
   583→   583→        isPredicted: false, isFertile: false,
   584→   584→      });
   585→   585→    }
   586→   586→
   587→   587→    for (let day = 1; day <= totalDays; day++) {
   588→   588→      const dateStr = `${calYear}-${String(calMonth).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
   589→   589→      const isToday = dateStr === todayStr;
   590→   590→      const periodInfo = getPeriodInfo(dateStr);
   591→   591→      const isPredicted = predictedDays.includes(dateStr) && !periodInfo.isPeriod;
   592→   592→      const isFertile = fertileDays.includes(dateStr) && !periodInfo.isPeriod && !isPredicted;
   593→   593→
   594→   594→      days.push({ day, dateStr, isToday, isOtherMonth: false, periodInfo, isPredicted, isFertile });
   595→   595→    }
   596→   596→
   597→   597→    return days;
   598→   598→  }
