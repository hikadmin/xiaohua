   316→  // ============ Period Logic ============
   317→  function getPeriodInfo(dateStr: string): PeriodInfoResult {
   318→    let isPeriod = false, isStart = false, isEnd = false, isActive = false;
   319→    for (const period of periods) {
   320→      if (period.startDate && !period.endDate) {
   321→        if (dateStr === period.startDate) { isPeriod = true; isStart = true; isEnd = true; isActive = true; break; }
   322→        const startD = parseDate(period.startDate);
   323→        const checkD = parseDate(dateStr);
   324→        if (checkD > startD && checkD <= new Date()) { isPeriod = true; if (dateStr === period.startDate) isStart = true; isActive = true; break; }
   325→      } else if (period.startDate && period.endDate) {
   326→        if (dateStr >= period.startDate && dateStr <= period.endDate) { isPeriod = true; if (dateStr === period.startDate) isStart = true; if (dateStr === period.endDate) isEnd = true; break; }
   327→      }
   328→    }
   329→    return { isPeriod, isStart, isEnd, isActive };
   330→  }

   336→  function getCycleInfo(): CycleInfo {
   337→    const cycleLength = profile?.cycleLength || 28;
   338→    const periodLength = profile?.periodLength || 5;
   339→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   340→    const lastPeriod = sortedPeriods[0];
   341→    if (!lastPeriod) return { phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength, cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: null };
   342→    const lastStart = parseDate(lastPeriod.startDate);
   343→    const now = new Date(); now.setHours(0, 0, 0, 0);
   344→    const daysSinceStart = daysBetween(lastStart, now);
   345→    if (daysSinceStart < 0) return { phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength, cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: lastPeriod.startDate };
   346→    const dayInCycle = (daysSinceStart % cycleLength) + 1;
   347→    const nextPeriodDate = addDays(lastStart, cycleLength);
   348→    const daysUntilNext = Math.max(0, daysBetween(now, nextPeriodDate));
   349→    let phase: string;
   350→    if (dayInCycle <= periodLength) phase = 'period';
   351→    else if (dayInCycle <= 13) phase = 'follicular';
   352→    else if (dayInCycle <= 16) phase = 'ovulation';
   353→    else phase = 'luteal';
   354→    return { phase, phaseDay: dayInCycle, daysUntilNext, cycleLength, periodLength, nextPeriodDate, lastPeriodStart: lastPeriod.startDate };
   355→  }

   357→  function getPredictedPeriodDays(): string[] {
   358→    const cycleLength = profile?.cycleLength || 28;
   359→    const periodLength = profile?.periodLength || 5;
   360→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   361→    const lastPeriod = sortedPeriods[0];
   362→    if (!lastPeriod || !lastPeriod.endDate) return [];
   363→    const lastEnd = parseDate(lastPeriod.endDate);
   364→    const nextStart = addDays(lastEnd, cycleLength - periodLength + 1);
   365→    return Array.from({ length: periodLength }, (_, i) => formatDateStr(addDays(nextStart, i)));
   366→  }
   367→
   368→  function getFertileDays(): string[] {
   369→    const cycleLength = profile?.cycleLength || 28;
   370→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   371→    const lastPeriod = sortedPeriods[0];
   372→    if (!lastPeriod) return [];
   373→    const lastStart = parseDate(lastPeriod.startDate);
   374→    const ovulationDay = addDays(lastStart, cycleLength - 14);
   375→    return Array.from({ length: 6 }, (_, i) => formatDateStr(addDays(ovulationDay, i - 4)));
   376→  }

Math.round(pl.reduce((a, b) => a + b, 0) / pl.length) : (profile?.periodLength || 5), totalCycles: sp.length, cycleLengths: cl, periodLengths: pl };
