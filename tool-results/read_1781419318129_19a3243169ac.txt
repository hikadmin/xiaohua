   165→  // ============ Period Logic (前端仍需用于日历渲染) ============
   166→  function getPeriodInfo(dateStr: string): PeriodInfoResult {
   167→    let isPeriod = false;
   168→    let isStart = false;
   169→    let isEnd = false;
   170→    let isActive = false;
   171→
   172→    for (const period of periods) {
   173→      if (period.startDate && !period.endDate) {
   174→        if (dateStr === period.startDate) {
   175→          isPeriod = true; isStart = true; isEnd = true; isActive = true;
   176→          break
   177→        }
   178→        const startD = parseDate(period.startDate);
   179→        const checkD = parseDate(dateStr);
   180→        if (checkD > startD && checkD <= new Date()) {
   181→          isPeriod = true;
   182→          if (dateStr === period.startDate) isStart = true;
   183→          isActive = true;
   184→          break
   185→        }
   186→      } else if (period.startDate && period.endDate) {
   187→        if (dateStr >= period.startDate && dateStr <= period.endDate) {
   188→          isPeriod = true;
   189→          if (dateStr === period.startDate) isStart = true;
   190→          if (dateStr === period.endDate) isEnd = true;
   191→          break
   192→        }
   193→      }
   194→    }
   195→    return { isPeriod, isStart, isEnd, isActive };
   196→  }

   202→  function getCycleInfo(): CycleInfo {
   203→    const cycleLength = profile?.cycleLength || 28;
   204→    const periodLength = profile?.periodLength || 5;
   205→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   206→    const lastPeriod = sortedPeriods[0];
   207→
   208→    if (!lastPeriod) {
   209→      return {
   210→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   211→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: null,
   212→      };
   213→    }
   214→
   215→    const lastStart = parseDate(lastPeriod.startDate);
   216→    const now = new Date();
   217→    now.setHours(0, 0, 0, 0);
   218→    const daysSinceStart = daysBetween(lastStart, now);
   219→
   220→    if (daysSinceStart < 0) {
   221→      return {
   222→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   223→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: lastPeriod.startDate,
   224→      };
   225→    }
   226→
   227→    const dayInCycle = (daysSinceStart % cycleLength) + 1;
   228→    const nextPeriodDate = addDays(lastStart, cycleLength);
   229→    const daysUntilNext = Math.max(0, daysBetween(now, nextPeriodDate));
   230→
   231→    let phase: string;
   232→    if (dayInCycle <= periodLength) {
   233→      phase = 'period';
   234→    } else if (dayInCycle <= 13) {
   235→      phase = 'follicular';
   236→    } else if (dayInCycle <= 16) {
   237→      phase = 'ovulation';
   238→    } else {
   239→      phase = 'luteal';
   240→    }
   241→
   242→    return {
   243→      phase, phaseDay: dayInCycle, daysUntilNext,
   244→      cycleLength, periodLength, nextPeriodDate, lastPeriodStart: lastPeriod.startDate,
   245→    };
   246→  }

   248→  function getPredictedPeriodDays(): string[] {
   249→    const cycleLength = profile?.cycleLength || 28;
   250→    const periodLength = profile?.periodLength || 5;
   251→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   252→    const lastPeriod = sortedPeriods[0];
   253→    if (!lastPeriod || !lastPeriod.endDate) return [];
   254→
   255→    const lastEnd = parseDate(lastPeriod.endDate);
   256→    const nextStart = addDays(lastEnd, cycleLength - periodLength + 1);
   257→    const days: string[] = [];
   258→    for (let i = 0; i < periodLength; i++) {
   259→      days.push(formatDateStr(addDays(nextStart, i)));
   260→    }
   261→    return days;
   262→  }

   264→  function getFertileDays(): string[] {
   265→    const cycleLength = profile?.cycleLength || 28;
   266→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   267→    const lastPeriod = sortedPeriods[0];
   268→    if (!lastPeriod) return [];
   269→
   270→    const lastStart = parseDate(lastPeriod.startDate);
   271→    const ovulationDay = addDays(lastStart, cycleLength - 14);
   272→    const days: string[] = [];
   273→    for (let i = -4; i <= 1; i++) {
   274→      days.push(formatDateStr(addDays(ovulationDay, i)));
   275→    }
   276→    return days;
   277→  }

   464→  // ============ Calendar Generation ============
   465→  function generateCalendarDays(): CalendarDay[] {
   466→    const firstDay = new Date(calYear, calMonth - 1, 1);
   467→    const lastDay = new Date(calYear, calMonth, 0);
   468→    const startDayOfWeek = firstDay.getDay();
   469→    const totalDays = lastDay.getDate();
   470→    const todayStr = formatDateStr(new Date());
   471→    const predictedDays = getPredictedPeriodDays();
   472→    const fertileDays = getFertileDays();
   473→
   474→    const days: CalendarDay[] = [];
   475→
   476→    for (let i = 0; i < startDayOfWeek; i++) {
   477→      days.push({
   478→        day: 0, dateStr: '', isToday: false, isOtherMonth: true,
   479→        periodInfo: { isPeriod: false, isStart: false, isEnd: false, isActive: false },
   480→        isPredicted: false, isFertile: false,
   481→      });
   482→    }
   483→
   484→    for (let day = 1; day <= totalDays; day++) {
   485→      const dateStr = `${calYear}-${String(calMonth).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
   486→      const isToday = dateStr === todayStr;
   487→      const periodInfo = getPeriodInfo(dateStr);
   488→      const isPredicted = predictedDays.includes(dateStr) && !periodInfo.isPeriod;
   489→      const isFertile = fertileDays.includes(dateStr) && !periodInfo.isPeriod && !isPredicted;
   490→
   491→      days.push({ day, dateStr, isToday, isOtherMonth: false, periodInfo, isPredicted, isFertile });
   492→    }
   493→
   494→    return days;
   495→  }

   497→  // ============ Cycle Statistics ============
   498→  const cycleStats: CycleStats = useMemo(() => {
   499→    const sortedPeriods = [...periods].sort((a, b) => a.startDate.localeCompare(b.startDate));
   500→    const cycleLengths: number[] = [];
   501→    const periodLengths: number[] = [];
   502→
   503→    for (let i = 1; i < sortedPeriods.length; i++) {
   504→      if (sortedPeriods[i].endDate) {
   505→        const cycleLen = daysBetween(parseDate(sortedPeriods[i - 1].startDate), parseDate(sortedPeriods[i].startDate));
   506→        if (cycleLen > 15 && cycleLen < 50) cycleLengths.push(cycleLen);
   507→      }
   508→      if (sortedPeriods[i].endDate) {
   509→        const periodLen = daysBetween(parseDate(sortedPeriods[i].startDate), parseDate(sortedPeriods[i].endDate!)) + 1;
   510→        if (periodLen > 0 && periodLen < 15) periodLengths.push(periodLen);
   511→      }
   512→    }
   513→
   514→    const avgCycle = cycleLengths.length > 0
   515→      ? Math.round(cycleLengths.reduce((a, b) => a + b, 0) / cycleLengths.length)
   516→      : (profile?.cycleLength || 28);
   517→    const avgPeriod = periodLengths.length > 0
   518→      ? Math.round(periodLengths.reduce((a, b) => a + b, 0) / periodLengths.length)
   519→      : (profile?.periodLength || 5);
   520→
   521→    return {
   522→      avgCycle,
   523→      avgPeriod,
   524→      totalCycles: sortedPeriods.length,
   525→      cycleLengths,
   526→      periodLengths,
   527→    };
   528→  }, [periods, profile]);
