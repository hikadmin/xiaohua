   197→  // ============ Period Logic (前端仍需用于日历渲染) ============
   198→  function getPeriodInfo(dateStr: string): PeriodInfoResult {
   199→    let isPeriod = false;
   200→    let isStart = false;
   201→    let isEnd = false;
   202→    let isActive = false;
   203→
   204→    for (const period of periods) {
   205→      if (period.startDate && !period.endDate) {
   206→        if (dateStr === period.startDate) {
   207→          isPeriod = true; isStart = true; isEnd = true; isActive = true;
   208→          break
   209→        }
   210→        const startD = parseDate(period.startDate);
   211→        const checkD = parseDate(dateStr);
   212→        if (checkD > startD && checkD <= new Date()) {
   213→          isPeriod = true;
   214→          if (dateStr === period.startDate) isStart = true;
   215→          isActive = true;
   216→          break
   217→        }
   218→      } else if (period.startDate && period.endDate) {
   219→        if (dateStr >= period.startDate && dateStr <= period.endDate) {
   220→          isPeriod = true;
   221→          if (dateStr === period.startDate) isStart = true;
   222→          if (dateStr === period.endDate) isEnd = true;
   223→          break
   224→        }
   225→      }
   226→    }
   227→    return { isPeriod, isStart, isEnd, isActive };
   228→  }

   234→  function getCycleInfo(): CycleInfo {
   235→    const cycleLength = profile?.cycleLength || 28;
   236→    const periodLength = profile?.periodLength || 5;
   237→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   238→    const lastPeriod = sortedPeriods[0];
   239→
   240→    if (!lastPeriod) {
   241→      return {
   242→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   243→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: null,
   244→      };
   245→    }
   246→
   247→    const lastStart = parseDate(lastPeriod.startDate);
   248→    const now = new Date();
   249→    now.setHours(0, 0, 0, 0);
   250→    const daysSinceStart = daysBetween(lastStart, now);
   251→
   252→    if (daysSinceStart < 0) {
   253→      return {
   254→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   255→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: lastPeriod.startDate,
   256→      };
   257→    }
   258→
   259→    const dayInCycle = (daysSinceStart % cycleLength) + 1;
   260→    const nextPeriodDate = addDays(lastStart, cycleLength);
   261→    const daysUntilNext = Math.max(0, daysBetween(now, nextPeriodDate));
   262→
   263→    let phase: string;
   264→    if (dayInCycle <= periodLength) {
   265→      phase = 'period';
   266→    } else if (dayInCycle <= 13) {
   267→      phase = 'follicular';
   268→    } else if (dayInCycle <= 16) {
   269→      phase = 'ovulation';
   270→    } else {
   271→      phase = 'luteal';
   272→    }
   273→
   274→    return {
   275→      phase, phaseDay: dayInCycle, daysUntilNext,
   276→      cycleLength, periodLength, nextPeriodDate, lastPeriodStart: lastPeriod.startDate,
   277→    };
   278→  }

   280→  function getPredictedPeriodDays(): string[] {
   281→    const cycleLength = profile?.cycleLength || 28;
   282→    const periodLength = profile?.periodLength || 5;
   283→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   284→    const lastPeriod = sortedPeriods[0];
   285→    if (!lastPeriod || !lastPeriod.endDate) return [];
   286→
   287→    const lastEnd = parseDate(lastPeriod.endDate);
   288→    const nextStart = addDays(lastEnd, cycleLength - periodLength + 1);
   289→    const days: string[] = [];
   290→    for (let i = 0; i < periodLength; i++) {
   291→      days.push(formatDateStr(addDays(nextStart, i)));
   292→    }
   293→    return days;
   294→  }

   296→  function getFertileDays(): string[] {
   297→    const cycleLength = profile?.cycleLength || 28;
   298→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   299→    const lastPeriod = sortedPeriods[0];
   300→    if (!lastPeriod) return [];
   301→
   302→    const lastStart = parseDate(lastPeriod.startDate);
   303→    const ovulationDay = addDays(lastStart, cycleLength - 14);
   304→    const days: string[] = [];
   305→    for (let i = -4; i <= 1; i++) {
   306→      days.push(formatDateStr(addDays(ovulationDay, i)));
   307→    }
   308→    return days;
   309→  }

   520→  // ============ Calendar Generation ============
   521→  function generateCalendarDays(): CalendarDay[] {
   522→    const firstDay = new Date(calYear, calMonth - 1, 1);
   523→    const lastDay = new Date(calYear, calMonth, 0);
   524→    const startDayOfWeek = firstDay.getDay();
   525→    const totalDays = lastDay.getDate();
   526→    const todayStr = formatDateStr(new Date());
   527→    const predictedDays = getPredictedPeriodDays();
   528→    const fertileDays = getFertileDays();
   529→
   530→    const days: CalendarDay[] = [];
   531→
   532→    for (let i = 0; i < startDayOfWeek; i++) {
   533→      days.push({
   534→        day: 0, dateStr: '', isToday: false, isOtherMonth: true,
   535→        periodInfo: { isPeriod: false, isStart: false, isEnd: false, isActive: false },
   536→        isPredicted: false, isFertile: false,
   537→      });
   538→    }
   539→
   540→    for (let day = 1; day <= totalDays; day++) {
   541→      const dateStr = `${calYear}-${String(calMonth).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
   542→      const isToday = dateStr === todayStr;
   543→      const periodInfo = getPeriodInfo(dateStr);
   544→      const isPredicted = predictedDays.includes(dateStr) && !periodInfo.isPeriod;
   545→      const isFertile = fertileDays.includes(dateStr) && !periodInfo.isPeriod && !isPredicted;
   546→
   547→      days.push({ day, dateStr, isToday, isOtherMonth: false, periodInfo, isPredicted, isFertile });
   548→    }
