   130тЖТ   130тЖТ  function getPeriodInfo(dateStr: string): PeriodInfoResult {
   131тЖТ   131тЖТ    let isPeriod = false;
   132тЖТ   132тЖТ    let isStart = false;
   133тЖТ   133тЖТ    let isEnd = false;
   134тЖТ   134тЖТ    let isActive = false;
   135тЖТ   135тЖТ
   136тЖТ   136тЖТ    for (const period of periods) {
   137тЖТ   137тЖТ      if (period.startDate && !period.endDate) {
   138тЖТ   138тЖТ        if (dateStr === period.startDate) {
   139тЖТ   139тЖТ          isPeriod = true; isStart = true; isEnd = true; isActive = true;
   140тЖТ   140тЖТ          break;
   141тЖТ   141тЖТ        }
   142тЖТ   142тЖТ        const startD = parseDate(period.startDate);
   143тЖТ   143тЖТ        const checkD = parseDate(dateStr);
   144тЖТ   144тЖТ        if (checkD > startD && checkD <= new Date()) {
   145тЖТ   145тЖТ          isPeriod = true;
   146тЖТ   146тЖТ          if (dateStr === period.startDate) isStart = true;
   147тЖТ   147тЖТ          isActive = true;
   148тЖТ   148тЖТ          break;
   149тЖТ   149тЖТ        }
   150тЖТ   150тЖТ      } else if (period.startDate && period.endDate) {
   151тЖТ   151тЖТ        if (dateStr >= period.startDate && dateStr <= period.endDate) {
   152тЖТ   152тЖТ          isPeriod = true;
   153тЖТ   153тЖТ          if (dateStr === period.startDate) isStart = true;
   154тЖТ   154тЖТ          if (dateStr === period.endDate) isEnd = true;
   155тЖТ   155тЖТ          break;
   156тЖТ   156тЖТ        }
   157тЖТ   157тЖТ      }
   158тЖТ   158тЖТ    }
   159тЖТ   159тЖТ    return { isPeriod, isStart, isEnd, isActive };
   160тЖТ   160тЖТ  }

   166тЖТ   166тЖТ  function getCycleInfo(): CycleInfo {
   167тЖТ   167тЖТ    const cycleLength = profile?.cycleLength || 28;
   168тЖТ   168тЖТ    const periodLength = profile?.periodLength || 5;
   169тЖТ   169тЖТ    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   170тЖТ   170тЖТ    const lastPeriod = sortedPeriods[0];
   171тЖТ   171тЖТ
   172тЖТ   172тЖТ    if (!lastPeriod) {
   173тЖТ   173тЖТ      return {
   174тЖТ   174тЖТ        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   175тЖТ   175тЖТ        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: null,
   176тЖТ   176тЖТ      };
   177тЖТ   177тЖТ    }
   178тЖТ   178тЖТ
   179тЖТ   179тЖТ    const lastStart = parseDate(lastPeriod.startDate);
   180тЖТ   180тЖТ    const now = new Date();
   181тЖТ   181тЖТ    now.setHours(0, 0, 0, 0);
   182тЖТ   182тЖТ    const daysSinceStart = daysBetween(lastStart, now);
   183тЖТ   183тЖТ
   184тЖТ   184тЖТ    if (daysSinceStart < 0) {
   185тЖТ   185тЖТ      return {
   186тЖТ   186тЖТ        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   187тЖТ   187тЖТ        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: lastPeriod.startDate,
   188тЖТ   188тЖТ      };
   189тЖТ   189тЖТ    }
   190тЖТ   190тЖТ
   191тЖТ   191тЖТ    const dayInCycle = (daysSinceStart % cycleLength) + 1;
   192тЖТ   192тЖТ    const nextPeriodDate = addDays(lastStart, cycleLength);
   193тЖТ   193тЖТ    const daysUntilNext = Math.max(0, daysBetween(now, nextPeriodDate));
   194тЖТ   194тЖТ
   195тЖТ   195тЖТ    let phase: string;
   196тЖТ   196тЖТ    if (dayInCycle <= periodLength) {
   197тЖТ   197тЖТ      phase = 'period';
   198тЖТ   198тЖТ    } else if (dayInCycle <= 13) {
   199тЖТ   199тЖТ      phase = 'follicular';
   200тЖТ   200тЖТ    } else if (dayInCycle <= 16) {
   201тЖТ   201тЖТ      phase = 'ovulation';
   202тЖТ   202тЖТ    } else {
   203тЖТ   203тЖТ      phase = 'luteal';
   204тЖТ   204тЖТ    }
   205тЖТ   205тЖТ
   206тЖТ   206тЖТ    return {
   207тЖТ   207тЖТ      phase, phaseDay: dayInCycle, daysUntilNext,
   208тЖТ   208тЖТ      cycleLength, periodLength, nextPeriodDate, lastPeriodStart: lastPeriod.startDate,
   209тЖТ   209тЖТ    };
   210тЖТ   210тЖТ  }

   212тЖТ   212тЖТ  function getPredictedPeriodDays(): string[] {
   213тЖТ   213тЖТ    const cycleLength = profile?.cycleLength || 28;
   214тЖТ   214тЖТ    const periodLength = profile?.periodLength || 5;
   215тЖТ   215тЖТ    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   216тЖТ   216тЖТ    const lastPeriod = sortedPeriods[0];
   217тЖТ   217тЖТ    if (!lastPeriod || !lastPeriod.endDate) return [];
   218тЖТ   218тЖТ
   219тЖТ   219тЖТ    const lastEnd = parseDate(lastPeriod.endDate);
   220тЖТ   220тЖТ    const nextStart = addDays(lastEnd, cycleLength - periodLength + 1);
   221тЖТ   221тЖТ    const days: string[] = [];
   222тЖТ   222тЖТ    for (let i = 0; i < periodLength; i++) {
   223тЖТ   223тЖТ      days.push(formatDateStr(addDays(nextStart, i)));
   224тЖТ   224тЖТ    }
   225тЖТ   225тЖТ    return days;
   226тЖТ   226тЖТ  }

   438тЖТ   438тЖТ    const predictedDays = getPredictedPeriodDays();
   439тЖТ   439тЖТ    const fertileDays = getFertileDays();
   440тЖТ   440тЖТ
   441тЖТ   441тЖТ    const days: CalendarDay[] = [];
   442тЖТ   442тЖТ
   443тЖТ   443тЖТ    for (let i = 0; i < startDayOfWeek; i++) {
   444тЖТ   444тЖТ      days.push({
   445тЖТ   445тЖТ        day: 0, dateStr: '', isToday: false, isOtherMonth: true,
   446тЖТ   446тЖТ        periodInfo: { isPeriod: false, isStart: false, isEnd: false, isActive: false },
