   181→  // ============ Period Logic (前端仍需用于日历渲染) ============
   182→  function getPeriodInfo(dateStr: string): PeriodInfoResult {
   183→    let isPeriod = false;
   184→    let isStart = false;
   185→    let isEnd = false;
   186→    let isActive = false;
   187→
   188→    for (const period of periods) {
   189→      if (period.startDate && !period.endDate) {
   190→        if (dateStr === period.startDate) {
   191→          isPeriod = true; isStart = true; isEnd = true; isActive = true;
   192→          break
   193→        }
   194→        const startD = parseDate(period.startDate);
   195→        const checkD = parseDate(dateStr);
   196→        if (checkD > startD && checkD <= new Date()) {
   197→          isPeriod = true;
   198→          if (dateStr === period.startDate) isStart = true;
   199→          isActive = true;
   200→          break
   201→        }
   202→      } else if (period.startDate && period.endDate) {
   203→        if (dateStr >= period.startDate && dateStr <= period.endDate) {
   204→          isPeriod = true;
   205→          if (dateStr === period.startDate) isStart = true;
   206→          if (dateStr === period.endDate) isEnd = true;
   207→          break
   208→        }
   209→      }
   210→    }
   211→    return { isPeriod, isStart, isEnd, isActive };
   212→  }

   218→  function getCycleInfo(): CycleInfo {
   219→    const cycleLength = profile?.cycleLength || 28;
   220→    const periodLength = profile?.periodLength || 5;
   221→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   222→    const lastPeriod = sortedPeriods[0];
   223→
   224→    if (!lastPeriod) {
   225→      return {
   226→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   227→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: null,
   228→      };
   229→    }
   230→
   231→    const lastStart = parseDate(lastPeriod.startDate);
   232→    const now = new Date();
   233→    now.setHours(0, 0, 0, 0);
   234→    const daysSinceStart = daysBetween(lastStart, now);
   235→
   236→    if (daysSinceStart < 0) {
   237→      return {
   238→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   239→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: lastPeriod.startDate,
   240→      };
   241→    }
   242→
   243→    const dayInCycle = (daysSinceStart % cycleLength) + 1;
   244→    const nextPeriodDate = addDays(lastStart, cycleLength);
   245→    const daysUntilNext = Math.max(0, daysBetween(now, nextPeriodDate));
   246→
   247→    let phase: string;
   248→    if (dayInCycle <= periodLength) {
   249→      phase = 'period';
   250→    } else if (dayInCycle <= 13) {
   251→      phase = 'follicular';
   252→    } else if (dayInCycle <= 16) {
   253→      phase = 'ovulation';
   254→    } else {
   255→      phase = 'luteal';
   256→    }
   257→
   258→    return {
   259→      phase, phaseDay: dayInCycle, daysUntilNext,
   260→      cycleLength, periodLength, nextPeriodDate, lastPeriodStart: lastPeriod.startDate,
   261→    };
   262→  }

   264→  function getPredictedPeriodDays(): string[] {
   265→    const cycleLength = profile?.cycleLength || 28;
   266→    const periodLength = profile?.periodLength || 5;
   267→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   268→    const lastPeriod = sortedPeriods[0];
   269→    if (!lastPeriod || !lastPeriod.endDate) return [];
   270→
   271→    const lastEnd = parseDate(lastPeriod.endDate);
   272→    const nextStart = addDays(lastEnd, cycleLength - periodLength + 1);
   273→    const days: string[] = [];
   274→    for (let i = 0; i < periodLength; i++) {
   275→      days.push(formatDateStr(addDays(nextStart, i)));
   276→    }
   277→    return days;
   278→  }

   280→  function getFertileDays(): string[] {
   281→    const cycleLength = profile?.cycleLength || 28;
   282→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   283→    const lastPeriod = sortedPeriods[0];
   284→    if (!lastPeriod) return [];
   285→
   286→    const lastStart = parseDate(lastPeriod.startDate);
   287→    const ovulationDay = addDays(lastStart, cycleLength - 14);
   288→    const days: string[] = [];
   289→    for (let i = -4; i <= 1; i++) {
   290→      days.push(formatDateStr(addDays(ovulationDay, i)));
   291→    }
   292→    return days;
   293→  }

   493→  // ============ Calendar Generation ============
   494→  function generateCalendarDays(): CalendarDay[] {
   495→    const firstDay = new Date(calYear, calMonth - 1, 1);
   496→    const lastDay = new Date(calYear, calMonth, 0);
   497→    const startDayOfWeek = firstDay.getDay();
   498→    const totalDays = lastDay.getDate();
   499→    const todayStr = formatDateStr(new Date());
   500→    const predictedDays = getPredictedPeriodDays();
   501→    const fertileDays = getFertileDays();
   502→
   503→    const days: CalendarDay[] = [];
   504→
   505→    for (let i = 0; i < startDayOfWeek; i++) {
   506→      days.push({
   507→        day: 0, dateStr: '', isToday: false, isOtherMonth: true,
   508→        periodInfo: { isPeriod: false, isStart: false, isEnd: false, isActive: false },
   509→        isPredicted: false, isFertile: false,
   510→      });
   511→    }
   512→
   513→    for (let day = 1; day <= totalDays; day++) {
   514→      const dateStr = `${calYear}-${String(calMonth).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
   515→      const isToday = dateStr === todayStr;
   516→      const periodInfo = getPeriodInfo(dateStr);
   517→      const isPredicted = predictedDays.includes(dateStr) && !periodInfo.isPeriod;
   518→      const isFertile = fertileDays.includes(dateStr) && !periodInfo.isPeriod && !isPredicted;
   519→
   520→      days.push({ day, dateStr, isToday, isOtherMonth: false, periodInfo, isPredicted, isFertile });
   521→    }
   522→
   523→    return days;
   524→  }

   526→  // ============ Cycle Statistics ============
   527→  const cycleStats: CycleStats = useMemo(() => {
   528→    const sortedPeriods = [...periods].sort((a, b) => a.startDate.localeCompare(b.startDate));
   529→    const cycleLengths: number[] = [];
   530→    const periodLengths: number[] = [];
   531→
   532→    for (let i = 1; i < sortedPeriods.length; i++) {
   533→      if (sortedPeriods[i].endDate) {
   534→        const cycleLen = daysBetween(parseDate(sortedPeriods[i - 1].startDate), parseDate(sortedPeriods[i].startDate));
   535→        if (cycleLen > 15 && cycleLen < 50) cycleLengths.push(cycleLen);
   536→      }
   537→      if (sortedPeriods[i].endDate) {
   538→        const periodLen = daysBetween(parseDate(sortedPeriods[i].startDate), parseDate(sortedPeriods[i].endDate!)) + 1;
   539→        if (periodLen > 0 && periodLen < 15) periodLengths.push(periodLen);
   540→      }
   541→    }
   542→
   543→    const avgCycle = cycleLengths.length > 0
