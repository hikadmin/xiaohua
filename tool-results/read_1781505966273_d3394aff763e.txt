   297→  // ============ Period Logic ============
   298→  function getPeriodInfo(dateStr: string): PeriodInfoResult {
   299→    let isPeriod = false, isStart = false, isEnd = false, isActive = false;
   300→    for (const period of periods) {
   301→      if (period.startDate && !period.endDate) {
   302→        if (dateStr === period.startDate) { isPeriod = true; isStart = true; isEnd = true; isActive = true; break; }
   303→        const startD = parseDate(period.startDate);
   304→        const checkD = parseDate(dateStr);
   305→        if (checkD > startD && checkD <= new Date()) { isPeriod = true; if (dateStr === period.startDate) isStart = true; isActive = true; break; }
   306→      } else if (period.startDate && period.endDate) {
   307→        if (dateStr >= period.startDate && dateStr <= period.endDate) { isPeriod = true; if (dateStr === period.startDate) isStart = true; if (dateStr === period.endDate) isEnd = true; break; }
   308→      }
   309→    }
   310→    return { isPeriod, isStart, isEnd, isActive };
   311→  }

   317→  function getCycleInfo(): CycleInfo {
   318→    const cycleLength = profile?.cycleLength || 28;
   319→    const periodLength = profile?.periodLength || 5;
   320→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   321→    const lastPeriod = sortedPeriods[0];
   322→    if (!lastPeriod) return { phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength, cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: null };
   323→    const lastStart = parseDate(lastPeriod.startDate);
   324→    const now = new Date(); now.setHours(0, 0, 0, 0);
   325→    const daysSinceStart = daysBetween(lastStart, now);
   326→    if (daysSinceStart < 0) return { phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength, cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: lastPeriod.startDate };
   327→    const dayInCycle = (daysSinceStart % cycleLength) + 1;
   328→    const nextPeriodDate = addDays(lastStart, cycleLength);
   329→    const daysUntilNext = Math.max(0, daysBetween(now, nextPeriodDate));
   330→    let phase: string;
   331→    if (dayInCycle <= periodLength) phase = 'period';
   332→    else if (dayInCycle <= 13) phase = 'follicular';
   333→    else if (dayInCycle <= 16) phase = 'ovulation';
   334→    else phase = 'luteal';
   335→    return { phase, phaseDay: dayInCycle, daysUntilNext, cycleLength, periodLength, nextPeriodDate, lastPeriodStart: lastPeriod.startDate };
   336→  }
   337→
   338→  function getPredictedPeriodDays(): string[] {
   339→    const cycleLength = profile?.cycleLength || 28;
   340→    const periodLength = profile?.periodLength || 5;
   341→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   342→    const lastPeriod = sortedPeriods[0];
   343→    if (!lastPeriod || !lastPeriod.endDate) return [];
   344→    const lastEnd = parseDate(lastPeriod.endDate);
   345→    const nextStart = addDays(lastEnd, cycleLength - periodLength + 1);
   346→    return Array.from({ length: periodLength }, (_, i) => formatDateStr(addDays(nextStart, i)));
   347→  }
   348→
   349→  function getFertileDays(): string[] {
   350→    const cycleLength = profile?.cycleLength || 28;
   351→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   352→    const lastPeriod = sortedPeriods[0];
   353→    if (!lastPeriod) return [];
   354→    const lastStart = parseDate(lastPeriod.startDate);
   355→    const ovulationDay = addDays(lastStart, cycleLength - 14);
   356→    return Array.from({ length: 6 }, (_, i) => formatDateStr(addDays(ovulationDay, i - 4)));
   357→  }

   430→  // ============ Calendar Generation ============
   431→  function generateCalendarDays(): CalendarDay[] {
   432→    const firstDay = new Date(calYear, calMonth - 1, 1);
   433→    const startDayOfWeek = firstDay.getDay();
   434→    const totalDays = new Date(calYear, calMonth, 0).getDate();
   435→    const todayStr = formatDateStr(new Date());
   436→    const predictedDays = getPredictedPeriodDays();
   437→    const fertileDays = getFertileDays();
   438→    const days: CalendarDay[] = [];
   439→    for (let i = 0; i < startDayOfWeek; i++) days.push({ day: 0, dateStr: '', isToday: false, isOtherMonth: true, periodInfo: { isPeriod: false, isStart: false, isEnd: false, isActive: false }, isPredicted: false, isFertile: false });
   440→    for (let day = 1; day <= totalDays; day++) {
   441→      const dateStr = `${calYear}-${String(calMonth).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
   442→      const periodInfo = getPeriodInfo(dateStr);
   443→      days.push({ day, dateStr, isToday: dateStr === todayStr, isOtherMonth: false, periodInfo, isPredicted: predictedDays.includes(dateStr) && !periodInfo.isPeriod, isFertile: fertileDays.includes(dateStr) && !periodInfo.isPeriod && !predictedDays.includes(dateStr) });
   444→    }
   445→    return days;
   446→  }
   447→
   448→  // ============ Cycle Statistics ============
   449→  const cycleStats: CycleStats = useMemo(() => {
   450→    const sp = [...periods].sort((a, b) => a.startDate.localeCompare(b.startDate));
   451→    const cl: number[] = [], pl: number[] = [];
   452→    for (let i = 1; i < sp.length; i++) {
   453→      // 周期长度只需两个相邻经期开始日期，不需要endDate
   454→      const c = daysBetween(parseDate(sp[i - 1].startDate), parseDate(sp[i].startDate));
   455→      if (c > 15 && c < 50) cl.push(c);
   456→      // 经期长度需要endDate
   457→      if (sp[i].endDate) { const p = daysBetween(parseDate(sp[i].startDate), parseDate(sp[i].endDate!)) + 1; if (p > 0 && p < 15) pl.push(p); }
   458→    }
   459→    return { avgCycle: cl.length > 0 ? Math.round(cl.reduce((a, b) => a + b, 0) / cl.length) : (profile?.cycleLength || 28), avgPeriod: pl.length > 0 ? Math.round(pl.reduce((a, b) => a + b, 0) / pl.length) : (profile?.periodLength || 5), totalCycles: sp.length, cycleLengths: cl, periodLengths: pl };
   460→  }, [periods, profile]);
   461→

   465→  const calendarDays = generateCalendarDays();

   479→        <div className="absolute rounded-full opacity-30 blur-[80px]" style={{ width: 300, height: 300, background: `linear-gradient(135deg, ${themeColor}40, transparent)`, top: -80, right: -80, animation: 'blob-float 20s ease-in-out infinite', transition: 'background 1s ease' }} />
   480→        <div className="absolute rounded-full opacity-25 blur-[60px]" style={{ width: 250, height: 250, background: 'linear-gradient(135deg, #81b29a, transparent)', bottom: 200, left: -80, animation: 'blob-float 20s ease-in-out infinite', animationDelay: '-7s' }} />
   481→        <div className="absolute rounded-full opacity-20 blur-[50px]" style={{ width: 180, height: 180, background: `linear-gradient(135deg, ${themeColor}80, transparent)`, bottom: -40, right: -40, animation: 'blob-float 20s ease-in-out infinite', animationDelay: '-14s' }} />
   482→      </div>
   483→
   484→      {/* Main Content */}
   485→      <main className="flex-1 relative z-10 overflow-y-auto overflow-x-hidden safe-top" style={{ paddingBottom: 90 }}>
   486→        <AnimatePresence mode="wait">
   487→          {activeTab === 'home' && <HomeTab today={today} cycleInfo={cycleInfo} cycleStats={cycleStats} records={records} dailyTipIndex={dailyTipIndex} setDailyTipIndex={setDailyTipIndex} setActiveTab={setActiveTab} setLogTab={setLogTab} ringAnimated={ringAnimated} notificationCount={notifications.length} onOpenNotification={() => setNotificationPanelOpen(true)} />}
   488→          {activeTab === 'calendar' && <CalendarTab calYear={calYear} calMonth={calMonth} setCalYear={setCalYear} setCalMonth={setCalMonth} isCurrentMonth={isCurrentMonth} today={today} calendarDays={calendarDays} periods={periods} setActionSheet={setActionSheet} />}
   489→          {activeTab === 'log' && <LogTab today={today} logTab={logTab} setLogTab={setLogTab} currentFlow={currentFlow} setCurrentFlow={setCurrentFlow} currentMood={currentMood} setCurrentMood={setCurrentMood} selectedSymptoms={selectedSymptoms} setSelectedSymptoms={setSelectedSymptoms} customSymptoms={customSymptoms} noteText={noteText} setNoteText={setNoteText} records={records} saveRecord={saveRecord} saveRecordForDate={saveRecordForDate} updateRecord={updateRecord} setSymptomSheetOpen={setSymptomSheetOpen} setDeleteConfirm={setDeleteConfirm} cycleInfo={cycleInfo} editingDate={editingDate} setEditingDate={setEditingDate} selectedDate={selectedDate} setSelectedDate={setSelectedDate} />}
   490→          {activeTab === 'profile' && <ProfileTab profile={profile} records={records} periods={periods} cycleStats={cycleStats} settings={settings} cycleInfo={cycleInfo} setProfileEditOpen={setProfileEditOpen} setEditName={setEditName} setEditAvatar={setEditAvatar} setEditCycleLength={setEditCycleLength} setEditPeriodLength={setEditPeriodLength} toggleSetting={toggleSetting} exportCSV={exportCSV} setFeedbackOpen={setFeedbackOpen} toast={toast} resetData={resetData} themeColor={themeColor} setThemeColor={setThemeColor} wallpaper={wallpaper} setWallpaper={setWallpaper} themeScope={themeScope} setThemeScope={setThemeScope} />}
   491→        </AnimatePresence>
   492→      </main>
   493→
   494→      {/* Bottom Navigation */}
   495→      <nav className="fixed bottom-0 left-0 right-0 z-50 safe-bottom" style={{ background: 'linear-gradient(to top, #0f1419 80%, transparent)' }}>
   496→        <div className="flex justify-around items-center pb-5 pt-2">
   497→          {[

   507→            <Plus size={26} style={{ color: '#0f1419', strokeWidth: 2.5 }} />

   514→              <item.icon size={22} style={{ color: activeTab === item.page ? themeColor : '#6b7280' }} />
