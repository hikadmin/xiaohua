   104→// ============ Utility Functions ============
   105→function formatDateStr(date: Date): string {
   106→  return `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}-${String(date.getDate()).padStart(2, '0')}`;
   107→}
   108→
   109→function formatDateChinese(date: Date): string {
   110→  return `${date.getMonth() + 1}月${date.getDate()}日 ${WEEKDAYS_FULL[date.getDay()]}`;
   111→}
   112→
   113→function formatShortDate(dateStr: string): string {
   114→  const parts = dateStr.split('-');
   115→  return `${parseInt(parts[1])}月${parseInt(parts[2])}日`;
   116→}
   117→
   118→function parseDate(dateStr: string): Date {
   119→  const parts = dateStr.split('-');
   120→  return new Date(parseInt(parts[0]), parseInt(parts[1]) - 1, parseInt(parts[2]));
   121→}
   122→
   123→function daysBetween(d1: Date, d2: Date): number {
   124→  return Math.round((d2.getTime() - d1.getTime()) / (1000 * 60 * 60 * 24));
   125→}
   126→
   127→function addDays(date: Date, days: number): Date {
   128→  const result = new Date(date);
   129→  result.setDate(result.getDate() + days);
   130→  return result;
   131→}

   243→  // ============ Period Logic ============
   244→  function getPeriodInfo(dateStr: string) {
   245→    let isPeriod = false;
   246→    let isStart = false;
   247→    let isEnd = false;
   248→    let isActive = false;
   249→
   250→    for (const period of periods) {
   251→      if (period.startDate && !period.endDate) {
   252→        if (dateStr === period.startDate) {
   253→          isPeriod = true; isStart = true; isEnd = true; isActive = true;
   254→          break;
   255→        }
   256→        // Active period extends from start to today
   257→        const startD = parseDate(period.startDate);
   258→        const checkD = parseDate(dateStr);
   259→        if (checkD > startD && checkD <= new Date()) {
   260→          isPeriod = true;
   261→          if (dateStr === period.startDate) isStart = true;
   262→          isActive = true;
   263→          break;
   264→        }
   265→      } else if (period.startDate && period.endDate) {
   266→        if (dateStr >= period.startDate && dateStr <= period.endDate) {
   267→          isPeriod = true;
   268→          if (dateStr === period.startDate) isStart = true;
   269→          if (dateStr === period.endDate) isEnd = true;
   270→          break;
   271→        }
   272→      }
   273→    }
   274→    return { isPeriod, isStart, isEnd, isActive };
   275→  }

   277→  function hasActivePeriod(): Period | null {

   281→  function getCycleInfo() {
   282→    const cycleLength = profile?.cycleLength || 28;
   283→    const periodLength = profile?.periodLength || 5;
   284→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   285→    const lastPeriod = sortedPeriods[0];
   286→
   287→    if (!lastPeriod) {
   288→      return {
   289→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   290→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: null,
   291→      };
   292→    }
   293→
   294→    const lastStart = parseDate(lastPeriod.startDate);
   295→    const now = new Date();
   296→    now.setHours(0, 0, 0, 0);
   297→    const daysSinceStart = daysBetween(lastStart, now);
   298→
   299→    if (daysSinceStart < 0) {
   300→      return {
   301→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   302→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: lastPeriod.startDate,
   303→      };
   304→    }
   305→
   306→    const dayInCycle = (daysSinceStart % cycleLength) + 1;
   307→    const nextPeriodDate = addDays(lastStart, cycleLength);
   308→    const daysUntilNext = Math.max(0, daysBetween(now, nextPeriodDate));
   309→
   310→    let phase: string;
   311→    if (dayInCycle <= periodLength) {
   312→      phase = 'period';
   313→    } else if (dayInCycle <= 13) {
   314→      phase = 'follicular';
   315→    } else if (dayInCycle <= 16) {
   316→      phase = 'ovulation';
   317→    } else {
   318→      phase = 'luteal';
   319→    }
   320→
   321→    return {
   322→      phase, phaseDay: dayInCycle, daysUntilNext,
   323→      cycleLength, periodLength, nextPeriodDate, lastPeriodStart: lastPeriod.startDate,
   324→    };
   325→  }
   326→
   327→  function getPredictedPeriodDays(): string[] {
   328→    const cycleLength = profile?.cycleLength || 28;
   329→    const periodLength = profile?.periodLength || 5;
   330→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   331→    const lastPeriod = sortedPeriods[0];
   332→    if (!lastPeriod || !lastPeriod.endDate) return [];
   333→
   334→    const lastEnd = parseDate(lastPeriod.endDate);
   335→    const nextStart = addDays(lastEnd, cycleLength - periodLength + 1);
   336→    const days: string[] = [];
   337→    for (let i = 0; i < periodLength; i++) {
   338→      days.push(formatDateStr(addDays(nextStart, i)));
   339→    }
   340→    return days;
   341→  }

   343→  function getFertileDays(): string[] {
   344→    const cycleLength = profile?.cycleLength || 28;

   346→    const lastPeriod = sortedPeriods[0];

   349→    const lastStart = parseDate(lastPeriod.startDate);

   474→  function exportCSV() {
   475→    const headers = ['日期', '流量', '情绪', '症状', '备注'];
   476→    const rows = records.map(r => {
   477→      const symptoms = JSON.parse(r.symptoms || '[]');

   481→        MOOD_LABELS[r.mood] || '',

   484→      ];
   485→    });

   487→    const BOM = '\uFEFF';
   488→    const blob = new Blob([BOM + csv], { type: 'text/csv;charset=utf-8;' });
   489→    const url = URL.createObjectURL(blob);
   490→    const a = document.createElement('a');
   491→    a.href = url;
   492→    a.download = `luna_records_${formatDateStr(new Date())}.csv`;
   493→    a.click();
   494→    URL.revokeObjectURL(url);
   495→    toast({ description: '数据导出成功 📁' });
   496→  }
