   106→// ============ Utility Functions ============
   107→function formatDateStr(date: Date): string {
   108→  return `${date.getFullYear()}-${String(date.getMonth() + 1).padStart(2, '0')}-${String(date.getDate()).padStart(2, '0')}`;
   109→}
   110→
   111→function formatDateChinese(date: Date): string {
   112→  return `${date.getMonth() + 1}月${date.getDate()}日 ${WEEKDAYS_FULL[date.getDay()]}`;
   113→}
   114→
   115→function formatShortDate(dateStr: string): string {
   116→  const parts = dateStr.split('-');
   117→  return `${parseInt(parts[1])}月${parseInt(parts[2])}日`;
   118→}
   119→
   120→function parseDate(dateStr: string): Date {
   121→  const parts = dateStr.split('-');
   122→  return new Date(parseInt(parts[0]), parseInt(parts[1]) - 1, parseInt(parts[2]));
   123→}
   124→
   125→function daysBetween(d1: Date, d2: Date): number {
   126→  return Math.round((d2.getTime() - d1.getTime()) / (1000 * 60 * 60 * 24));
   127→}
   128→
   129→function addDays(date: Date, days: number): Date {
   130→  const result = new Date(date);
   131→  result.setDate(result.getDate() + days);
   132→  return result;
   133→}

   256→  // ============ Period Logic ============
   257→  function getPeriodInfo(dateStr: string) {
   258→    let isPeriod = false;
   259→    let isStart = false;
   260→    let isEnd = false;
   261→    let isActive = false;
   262→
   263→    for (const period of periods) {
   264→      if (period.startDate && !period.endDate) {
   265→        if (dateStr === period.startDate) {
   266→          isPeriod = true; isStart = true; isEnd = true; isActive = true;
   267→          break;
   268→        }
   269→        // Active period extends from start to today
   270→        const startD = parseDate(period.startDate);
   271→        const checkD = parseDate(dateStr);
   272→        if (checkD > startD && checkD <= new Date()) {
   273→          isPeriod = true;
   274→          if (dateStr === period.startDate) isStart = true;
   275→          isActive = true;
   276→          break;
   277→        }
   278→      } else if (period.startDate && period.endDate) {
   279→        if (dateStr >= period.startDate && dateStr <= period.endDate) {
   280→          isPeriod = true;
   281→          if (dateStr === period.startDate) isStart = true;
   282→          if (dateStr === period.endDate) isEnd = true;
   283→          break;
   284→        }
   285→      }
   286→    }
   287→    return { isPeriod, isStart, isEnd, isActive };
   288→  }

   290→  function hasActivePeriod(): Period | null {

   294→  function getCycleInfo() {
   295→    const cycleLength = profile?.cycleLength || 28;
   296→    const periodLength = profile?.periodLength || 5;
   297→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   298→    const lastPeriod = sortedPeriods[0];
   299→
   300→    if (!lastPeriod) {
   301→      return {
   302→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   303→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: null,
   304→      };
   305→    }
   306→
   307→    const lastStart = parseDate(lastPeriod.startDate);
   308→    const now = new Date();
   309→    now.setHours(0, 0, 0, 0);
   310→    const daysSinceStart = daysBetween(lastStart, now);
   311→
   312→    if (daysSinceStart < 0) {
   313→      return {
   314→        phase: 'follicular', phaseDay: 1, daysUntilNext: cycleLength,
   315→        cycleLength, periodLength, nextPeriodDate: null, lastPeriodStart: lastPeriod.startDate,
   316→      };
   317→    }
   318→
   319→    const dayInCycle = (daysSinceStart % cycleLength) + 1;
   320→    const nextPeriodDate = addDays(lastStart, cycleLength);
   321→    const daysUntilNext = Math.max(0, daysBetween(now, nextPeriodDate));
   322→
   323→    let phase: string;
   324→    if (dayInCycle <= periodLength) {
   325→      phase = 'period';
   326→    } else if (dayInCycle <= 13) {
   327→      phase = 'follicular';
   328→    } else if (dayInCycle <= 16) {
   329→      phase = 'ovulation';
   330→    } else {
   331→      phase = 'luteal';
   332→    }
   333→
   334→    return {
   335→      phase, phaseDay: dayInCycle, daysUntilNext,
   336→      cycleLength, periodLength, nextPeriodDate, lastPeriodStart: lastPeriod.startDate,
   337→    };
   338→  }
   339→
   340→  function getPredictedPeriodDays(): string[] {
   341→    const cycleLength = profile?.cycleLength || 28;
   342→    const periodLength = profile?.periodLength || 5;
   343→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   344→    const lastPeriod = sortedPeriods[0];
   345→    if (!lastPeriod || !lastPeriod.endDate) return [];
   346→
   347→    const lastEnd = parseDate(lastPeriod.endDate);
   348→    const nextStart = addDays(lastEnd, cycleLength - periodLength + 1);
   349→    const days: string[] = [];
   350→    for (let i = 0; i < periodLength; i++) {
   351→      days.push(formatDateStr(addDays(nextStart, i)));
   352→    }
   353→    return days;
   354→  }

   356→  function getFertileDays(): string[] {
   357→    const cycleLength = profile?.cycleLength || 28;
   358→    const sortedPeriods = [...periods].sort((a, b) => b.startDate.localeCompare(a.startDate));
   359→    const lastPeriod = sortedPeriods[0];
   360→    if (!lastPeriod) return [];
   361→
   362→    const lastStart = parseDate(lastPeriod.startDate);
   363→    const ovulationDay = addDays(lastStart, cycleLength - 14);
   364→    const days: string[] = [];
   365→    for (let i = -4; i <= 1; i++) {
   366→      days.push(formatDateStr(addDays(ovulationDay, i)));
   367→    }
   368→    return days;
   369→  }

   560→  function generateCalendarDays() {
   561→    const firstDay = new Date(calYear, calMonth - 1, 1);
   562→    const lastDay = new Date(calYear, calMonth, 0);
   563→    const startDayOfWeek = firstDay.getDay();
   564→    const totalDays = lastDay.getDate();
   565→    const todayStr = formatDateStr(new Date());
   566→    const predictedDays = getPredictedPeriodDays();
   567→    const fertileDays = getFertileDays();
   568→
   569→    const days: Array<{
   570→      day: number;
   571→      dateStr: string;
   572→      isToday: boolean;
   573→      isOtherMonth: boolean;
   574→      periodInfo: ReturnType<typeof getPeriodInfo>;
   575→      isPredicted: boolean;
   576→      isFertile: boolean;
   577→    }> = [];
   578→
   579→    for (let i = 0; i < startDayOfWeek; i++) {
   580→      days.push({
   581→        day: 0, dateStr: '', isToday: false, isOtherMonth: true,
   582→        periodInfo: { isPeriod: false, isStart: false, isEnd: false, isActive: false },
   583→        isPredicted: false, isFertile: false,
   584→      });
   585→    }
   586→
   587→    for (let day = 1; day <= totalDays; day++) {
   588→      const dateStr = `${calYear}-${String(calMonth).padStart(2, '0')}-${String(day).padStart(2, '0')}`;
   589→      const isToday = dateStr === todayStr;
   590→      const periodInfo = getPeriodInfo(dateStr);
   591→      const isPredicted = predictedDays.includes(dateStr) && !periodInfo.isPeriod;
   592→      const isFertile = fertileDays.includes(dateStr) && !periodInfo.isPeriod && !isPredicted;
   593→
   594→      days.push({ day, dateStr, isToday, isOtherMonth: false, periodInfo, isPredicted, isFertile });
   595→    }
   596→
   597→    return days;
   598→  }
